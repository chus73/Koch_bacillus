22def read_functions_file(folder, file_name): 

23 """ 

24 Read the tb_functions.pl file and split the classes and functions ORF. 

25 :param file_name: The file name. 

26 :param folder: Folder which contain the file. 

27 :return: Two dictionaries, classes and functions from file. 

28 """ 

29 classes = {} 

30 functions = {} 

31 data = [] 

32 with open(os.path.join(folder, file_name)) as f: 

33 for line in f: 

34 if lookup_regex(line, '^class'): 

35 line = re.search("\((.*)\)", line).group(1) 

36 data.append(re.search("^\[.*\]", line).group(0)) 

37 data.append(re.search('".*', line).group(0)) 

38 # data[0]: Class Id 

39 # data[1]: Class description 

40 classes[data[0]] = data[1].replace('"', '') 

41 data = [] 

42 if lookup_regex(line, '^function'): 

43 line = re.search("\((.*)\)", line).group(1) 

44 regex = '(^\w*),(\[.*]),(''.*''),(".*")' 

45 result = re.search(regex, line) 

46 for i in range(len(result.groups())): 

47 if i == 0: 

48 key = result.groups()[0] 

49 else: 

50 data.append(result.groups()[i]) 

51 functions[key] = data 

52 for v in functions.values(): 

53 for d in v: 

54 d.replace("'", "") 

55 data = [] 

56 return classes, functions
